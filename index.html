%components chosen for optical fiber link of overall length of 10km and
%operating at 20Mbps 1. LED capable of launching an average power if 0.1mW
%at 0.85um diameter of core is 50um with graded index fiber, 2. fiber
%attenuation 2.5db/km, 3. requires splicing every 2Km with loss of 0.3db per
%splice. there is connection loss at the receiver of 1.5db, 4. the receiver
%requires mean incidenat optical power of -46dbm, 5. predicted safety
%margin of 6db. determine system is viable or not viable.
clc;
close all;
clear;
Len = input('Enter the length of the Link'); %length of fiber
Pa = input('Enter the Average Power'); %Ps in watt
LF = input('Enter the Fiber attenuation'); % fiber attenuation per km
n = input('Enter the distance between 2 splices'); % splices distance
spl = input('Enter the splice loss in system'); %loss per km
Lcn =input('Enter the connector loss');
Pr = input('Enter the mean incident power'); %pr receiver sensitivity
SM = input('Enter the Safety Margin'); %safety margin or system margin
Ps= 10*log10(Pa/1);
disp('Average power in dBm is')
disp(Ps)
Pt = Ps - Pr;
disp('allowed loss is');
disp(Pt);
Lc = LF*Len;
disp('Cable loss is');
disp(Lc);
S = (Len/n)-1;
disp('number of splices required are');
disp(S);
Ls = S*spl;
disp('splice loss is');
disp(Ls);
TL = Lc+Ls+(2*Lcn)+SM;
disp('Total Loss is:');
disp(TL);
if TL< Pt
disp('This Budget is Viable');
else
disp('This budget is not viable')
end
